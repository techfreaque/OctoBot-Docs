# timeframe

{% hint style="success" %}
the enabled triggering timeframe in your profile is used if you dont specify a timeframe
{% endhint %}

{% hint style="success" %}
the timeframe needs to be enabled in your profile
{% endhint %}

all_closes = await Close(ctx, time_frame="4h")
